// For demo we generate a URL or code that your bot server would also accept const pairingData = `pair:${sessionId}`; // replace with real pairing token from your bot // create QR image data URL try{ const qrDataUrl = await QRCode.toDataURL(pairingData, { errorCorrectionLevel: 'H' }); // store minimal session pairSessions.set(sessionId, { qr: pairingData, created: Date.now(), state: 'pending' }); // emit to requesting socket a pairing object socket.emit('pairing-ready', { sessionId, qrDataUrl, code: sessionId.match(/.{1,4}/g).join('-') }); // (Optional) set a TTL setTimeout(()=>{ if(pairSessions.has(sessionId)) pairSessions.delete(sessionId) }, 5*60*1000); }catch(err){ console.error('QR gen failed', err); socket.emit('pairing-error', { message: 'QR generation failed' }); } return res.status(200).json({ ok: true }); }); // Contact form endpoint app.post('/contact', (req, res) => { const { name, email, message } = req.body || {}; console.log('Contact form submitted', { name, email, message }); // TODO: Save to DB or forward to email via SMTP / third-party (SendGrid, Mailgun) return res.json({ ok: true, message: 'Thanks — we will contact you soon.' }); }); // Stripe checkout sample (server creates session) app.post('/create-checkout-session', async (req, res) => { // This requires STRIPE_SECRET in env const stripeSecret = process.env.STRIPE_SECRET;
